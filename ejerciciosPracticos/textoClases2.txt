#include <stdio.h>
#include <math.h>

int main() {

double saldo, monto , tasa, rendimiento;
int opcion, meses;

 printf("ingrese su saldo inicia: ");
 scanf("%lf", &saldo);
  do{
     printf("\n===== CAJERO BASICO ====\n");
      printf("Su saldo es: %.2f \n", saldo);
       printf("1) Depositar \n");
        printf("2) Retirar \n");
         printf("3) Invertir \n");
          printf("4) Salir \n");
          printf("Elija una opcion: ");
           scanf("%d", &opcion);

           switch (opcion)
           {
           case 1:
           printf("Monto a depositar: ");
           scanf("%lf", &monto);

           if(monto <= 0){
            printf("el monto debe ser mayor a 0. \n");
            }else {
              saldo += monto;
              printf("Deposito realizado. Nuevo saldo: %.2f\n", saldo);
            }
            break;
            case 2:
            printf("Coloque el monto a retirar: ");
            scanf("%lf", &monto);
            if(monto <= 0){
              printf("El monto debe ser mayor a 0. \n");
            }else if(monto>saldo){
              printf("Fondos insuficientes. \n");
            }else{
              saldo-= monto;
              printf("El retiro a sido realizado. Nuevo saldo: %.2f\n", saldo);
            }
            break;
            case 3:
            printf("Tiempo(meses): ");
            scanf("%d", &meses);

            printf("Tasa mensual (%%): ");
            scanf("%lf", &tasa);

            if(meses <= 0 || tasa<0){
              printf("Tiempo debe ser mayor o igual a 1 mes y la tasa debe ser positiva. \n");
            }else{
              rendimiento = saldo * (tasa/100.0)* meses;
              printf("rendimiento proyectado: %.2f\n", rendimiento);
               printf("saldo + rendimiento %.2f\n", saldo + rendimiento);
            }
            break;

            case 4:
            printf("Saliendo...\n");
            break;

            default:
            printf("Opcion invalida. Elija del 1 al 4. \n");
            break;
          }
          
        }while(opcion != 4);
          
          return 0;  

}
